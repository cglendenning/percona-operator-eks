kubectl -n NS run s3-copy-verify --rm -i --restart=Never \
  --image=amazon/aws-cli \
  --env="AWS_ACCESS_KEY_ID=..." \
  --env="AWS_SECRET_ACCESS_KEY=..." \
  --env="AWS_DEFAULT_REGION=us-east-1" \
  --command -- bash -lc '
set -euo pipefail

KEY="KEY"   # e.g. path/to/sst_info.000000...
SRC_EP="https://S3_GATEWAY_A"
DST_EP="https://S3_GATEWAY_B"
BUCKET="BUCKET"

echo "== Download from source A =="
aws s3 cp --endpoint-url "${SRC_EP}" "s3://${BUCKET}/${KEY}" /tmp/src.bin
wc -c /tmp/src.bin
sha256sum /tmp/src.bin | tee /tmp/src.sha

echo "== Upload to target B =="
aws s3 cp --endpoint-url "${DST_EP}" /tmp/src.bin "s3://${BUCKET}/${KEY}"

echo "== Download back from target B =="
aws s3 cp --endpoint-url "${DST_EP}" "s3://${BUCKET}/${KEY}" /tmp/dst.bin
wc -c /tmp/dst.bin
sha256sum /tmp/dst.bin | tee /tmp/dst.sha

echo "== Compare hashes =="
diff -u /tmp/src.sha /tmp/dst.sha && echo "HASH MATCH"

echo "== Last-byte range read on B =="
aws s3api head-object --endpoint-url "${DST_EP}" --bucket "${BUCKET}" --key "${KEY}" | sed -n "s/.*\"ContentLength\": \\([0-9]*\\).*/ContentLength=\\1/p"
aws s3api get-object --endpoint-url "${DST_EP}" --bucket "${BUCKET}" --key "${KEY}" /tmp/last1 --range bytes=146-146 >/dev/null
wc -c /tmp/last1
'
